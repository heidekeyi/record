     //typedef int ITEM_TYPE;
     typedef struct {
	 void *handle;
     } List;

     /*	explain return value follow	*/
     /*	0: call success	*/
     /*	-111: parameter error	*/
     /*	-333: alloc memory error	*/

     #define LIST_INTERFACE_GENERIC(ITEM_TYPE, SUFFIX_NAME) \
     \
     /*	init list env	*/\
     int list_create_##SUFFIX_NAME(List **list);\
     \
     /*	free resource of list	*/\
     int list_destroy_##SUFFIX_NAME(List **list);\
     \
     /*	retrieve element number of list	*/\
     int list_size_##SUFFIX_NAME(List *list, int *out_size);\
     \
     /*	retrieve index of element of list	*/\
     /*	-1: there not are match element	*/\
     int list_search_##SUFFIX_NAME(List *list,\
			 ITEM_TYPE in_data,\
			 int *out_index,\
			 int (*is_match)(ITEM_TYPE in_data, ITEM_TYPE ele_data));\
     \
     /*	insert new element	*/\
     /*	explain index follow	*/\
     /*	negative and out range: insert to last element	*/\
     /*	other: move original to next, insert to position of original	*/\
     int list_insert_##SUFFIX_NAME(List *list,\
			 int in_index,\
			 ITEM_TYPE in_data,\
			 int *out_index);\
     \
     /*	remove element	*/\
     /*	check index always	*/\
     /*	explain parameter	*/\
     /*	index is error when list is empty and out range, code is -111	*/\
     /*	save data if needed	*/\
     /*	free resource of element	*/\
     int list_remove_##SUFFIX_NAME(List *list, int in_index, ITEM_TYPE *out_data);\
     \
     /*	peek element	*/\
     /*	check index which the same of remove	*/\
     /*	save index if needed	*/\
     int list_peek_##SUFFIX_NAME(List *list, int in_index, ITEM_TYPE *out_data);\
     \
     /*	use new data replace original data	*/\
     /*	check index which the same of remove	*/\
     int list_replace_##SUFFIX_NAME(List *list,\
			  int in_index,\
			  ITEM_TYPE in_data,\
			  ITEM_TYPE *out_data);\
     \
     /*	reverse list	*/\
     int list_reverse_##SUFFIX_NAME(List *list);\
