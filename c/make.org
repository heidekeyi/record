** 为什么使用括号
   把一个字符当变量解析
   #+begin_src makefile
     v = variable                                                           
     e:
	     echo $vzzzzzz
   #+end_src
   #+begin_src shell
     [heidekeyi@localhost lib]$ make
     echo variablezzzzzz
     variablezzzzzz
   #+end_src
** 使用
   用()和{}都可以,等效的
   #+begin_src makefile
     v = variable
     e:
     echo ${v}
     echo $(v)
   #+end_src
   #+begin_src shell
     [heidekeyi@localhost lib]$ make
     echo variable
     variable
     echo variable
     variable
   #+end_src
